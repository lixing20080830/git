
这是个规则引擎Drools的例子，下面是git操作的常用命令

git init //把这个目录变成Git可以管理的仓库

git add README.md //文件添加到仓库

git add . //不但可以跟单一文件，还可以跟通配符，更可以跟目录。一个点就把当前目录下所有未追踪的文件全部add了 

git commit -m "first commit" //把文件提交到仓库

git remote rm origin //删除链接 

git remote add origin https://github.com/lixing20080830/sharding-jdbc-mybatis.git  //关联远程仓库
 
git push -u origin master -f //把本地库的所有内容强行推送到远程库上

git push -u origin master //把本地库的所有内容推送到远程库上


git status //查看整体变更（详细到文件名）

git log //查看历史记录
